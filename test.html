function runWikiLinkResolverTests() {
  if (typeof WikiLinkResolver === 'undefined') {
    assert(false, 'WikiLinkResolver: module exists');
    return;
  }
  assert(true, 'WikiLinkResolver: module exists');

  // Build a mock index
  const index = new Map();
  index.set('laplace transforms', { title: 'Laplace Transforms', path: 'warm/laplace-transforms.md' });
  index.set('fourier series', { title: 'Fourier Series', path: 'bedrock/fourier-series.md' });

  const resolver = WikiLinkResolver.create(index, 'Other Note');

  // Valid resolution
  const r1 = resolver.resolve('Laplace Transforms');
  assert(r1 !== null, 'WikiLinkResolver: resolves valid title');
  assert(r1.tagName === 'A', 'WikiLinkResolver: resolved link is an  element');

  // Case-insensitive resolution
  const r2 = resolver.resolve('laplace transforms');
  assert(r2 !== null, 'WikiLinkResolver: case-insensitive resolution works');

  // Missing title → broken link
  const r3 = resolver.resolve('Nonexistent Topic');
  assert(r3 !== null, 'WikiLinkResolver: missing title returns element (not null)');
  assert(r3.tagName !== 'A', 'WikiLinkResolver: missing title does not produce ');

  // Path traversal: ..
  const r4 = resolver.resolve('../../etc/passwd');
  assert(r4 === null || r4.tagName !== 'A', 'WikiLinkResolver: .. traversal rejected');

  // Path traversal: /
  const r5 = resolver.resolve('/etc/passwd');
  assert(r5 === null || r5.tagName !== 'A', 'WikiLinkResolver: absolute path rejected');

  // Path traversal: backslash
  const r6 = resolver.resolve('..\\..\\windows\\system32');
  assert(r6 === null || r6.tagName !== 'A', 'WikiLinkResolver: backslash traversal rejected');

  // Path traversal: NUL byte
  const r7 = resolver.resolve('test\0evil');
  assert(r7 === null || r7.tagName !== 'A', 'WikiLinkResolver: NUL byte rejected');

  // Path traversal: URL-encoded
  const r8 = resolver.resolve('%2e%2e%2fetc%2fpasswd');
  assert(r8 === null || r8.tagName !== 'A', 'WikiLinkResolver: URL-encoded traversal rejected');

  // Path traversal: tilde home
  const r9 = resolver.resolve('~/.ssh/id_rsa');
  assert(r9 === null || r9.tagName !== 'A', 'WikiLinkResolver: tilde path rejected');

  // Self-link → plain text
  const selfResolver = WikiLinkResolver.create(index, 'Laplace Transforms');
  const r10 = selfResolver.resolve('Laplace Transforms');
  assert(r10 === null || r10.tagName !== 'A', 'WikiLinkResolver: self-link does not produce ');
}

function runIndexRegeneratorTests() {
  if (typeof IndexRegenerator === 'undefined') {
    assert(false, 'IndexRegenerator: module exists');
    return;
  }
  assert(true, 'IndexRegenerator: module exists');

  // Empty notes array → header only
  const empty = IndexRegenerator.generate([]);
  assert(empty.includes('# Index'), 'IndexRegenerator: empty notes produces header');
  assert(empty.includes('| Title | Tier | Path | Updated |'), 'IndexRegenerator: empty notes produces table header');

  // Single note
  const one = IndexRegenerator.generate([{
    frontmatter: { title: 'Test Note', tier: 'warm', updated: '2026-04-21T14:22:00Z' },
    path: 'warm/test-note.md'
  }]);
  assert(one.includes('Test Note'), 'IndexRegenerator: includes note title');
  assert(one.includes('warm'), 'IndexRegenerator: includes note tier');
  assert(one.includes('warm/test-note.md'), 'IndexRegenerator: includes note path');

  // Idempotency
  const notes = [
    { frontmatter: { title: 'B Note', tier: 'cold', updated: '2026-04-20T00:00:00Z' }, path: 'cold/b-note.md' },
    { frontmatter: { title: 'A Note', tier: 'warm', updated: '2026-04-21T00:00:00Z' }, path: 'warm/a-note.md' }
  ];
  const run1 = IndexRegenerator.generate(notes);
  const run2 = IndexRegenerator.generate(notes);
  assertEqual(run1, run2, 'IndexRegenerator: idempotent (two runs produce identical output)');

  // Sort order: A before B
  const aIdx = run1.indexOf('A Note');
  const bIdx = run1.indexOf('B Note');
  assert(aIdx < bIdx, 'IndexRegenerator: notes sorted by title (A before B)');
}

function runBacklinkGraphTests() {
  if (typeof BacklinkGraph === 'undefined') {
    assert(false, 'BacklinkGraph: module exists');
    return;
  }
  assert(true, 'BacklinkGraph: module exists');

  const notes = [
    { path: 'warm/a.md', frontmatter: { title: 'Note A' }, wikiLinks: ['Note B', 'Note C'] },
    { path: 'warm/b.md', frontmatter: { title: 'Note B' }, wikiLinks: ['Note A'] },
    { path: 'cold/c.md', frontmatter: { title: 'Note C' }, wikiLinks: [] }
  ];

  const graph = BacklinkGraph.build(notes);

  // Note A is linked from Note B
  const aBacklinks = graph.get('warm/a.md') || [];
  assert(aBacklinks.some(b => b.path === 'warm/b.md'), 'BacklinkGraph: Note A has backlink from Note B');

  // Note B is linked from Note A
  const bBacklinks = graph.get('warm/b.md') || [];
  assert(bBacklinks.some(b => b.path === 'warm/a.md'), 'BacklinkGraph: Note B has backlink from Note A');

  // Note C is linked from Note A
  const cBacklinks = graph.get('cold/c.md') || [];
  assert(cBacklinks.some(b => b.path === 'warm/a.md'), 'BacklinkGraph: Note C has backlink from Note A');

  // Note C has no outgoing links so it doesn't appear as a backlink for others
  assert(!aBacklinks.some(b => b.path === 'cold/c.md'), 'BacklinkGraph: Note C does not backlink to Note A');
}
